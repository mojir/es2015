"use strict";
console.log("ES6 modules");

// Import all exports from module
import * as math from "./lib/math-utils";

let x = math.double(2); // -> 4
x = math.tripple(x); // -> 12
console.log(`Now x is ${x}`);

lib/fib.js

export default function fib(n) {
    if (n <= 0) {
        return NaN;
    }
    if (n <= 2) {
        return 1;
    }
    return fib(n - 1) + fib(n - 2);
}

lib/math-utils.js

// Export a function
export function double(x) {
    return 2 * x;
}


// Export another function
function tripple(x) {
    return 3 * x;
}
export {tripple};


// Export a constant
const PI = 3.14159;
export {PI};

lib/string-utils.js

"use strict";

export function surroundWithDashes(str) {
    return `---${str}---`;
}

export function surroundWithStars(str) {
    return `***${str}***`;
}
